function format(num){
	num += '';
	if (!num.includes('.')) num += '.';
	return num.replace(/(\d)(?=(\d{3})+\.)/g,function($0,$1){
		return $1 + ',';
	}).replace(/\.$/,'');
}
console.log(format('123456789.0'))

function sumString(a,b){
	var res = '', c = 0;
	a = a.split('');
	b = b.split('');

	while (a.length || b.length || c){
		c += ~~a.pop() + ~~b.pop();
		res = c % 10 + res;
		c = c>9
	}
	return res.replace(/^0+/,'');
}
